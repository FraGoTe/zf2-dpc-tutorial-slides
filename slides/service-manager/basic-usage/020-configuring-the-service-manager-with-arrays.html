Configuring the Service Manager with arrays:

$config = [
    'services' => [...],
    'invokables' => [...],
    'factories' => [...],
    'abstract_factories' => [...],
    'initializers' => [...],
    'delegators' => [...],
    'shared' => [...],
];

$serviceManager = new Zend\ServiceManager\ServiceManager(
    new Zend\ServiceManager\Config($config);
);
