application.config.php

Tells the whole application how to behave and which modules to load

return [
    'modules' => [
        'Application',
        'ZfcBase',
        'ZfcUser',
    ],
    'module_listener_options' => [
        'module_paths' => [
            './module',
            './vendor',
        ],
        'config_glob_paths' => [
            'config/autoload/{,*.}{global,local}.php',
        ],
    ],
    ...
];
